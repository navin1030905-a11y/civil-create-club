async function loadActivityLogs() { const box = document.getElementById("activityLogsBox"); if (adminRole !== "head") return; try { const res = await fetch(`${API_BASE}/activity-logs`, { headers: { authorization: "Bearer " + token } }); const data = await res.json(); if (!Array.isArray(data) || data.length === 0) { box.innerHTML = `
No activity logs found.
`; return; } box.innerHTML = data.map(log => `

${log.admin_username} ${log.admin_role}

Action: ${log.action_type}

${log.action_details || ""}

Time: ${new Date(log.created_at).toLocaleString()}

`).join(""); } catch (error) { box.innerHTML = `
Failed to load activity logs.
`; } }
